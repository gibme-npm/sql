Interface IDatabase

The core interface for all database types

interface IDatabase {
    close: (() => Promise<void>);
    createIndex: ((table, fields, type) => Promise<void>);
    createTable?: ((name, fields, primaryKey, tableOptions?, useTransaction?) => Promise<void>);
    dropTable: ((tables) => Promise<QueryResult[]>);
    escape: ((value) => string);
    escapeId: ((id) => string);
    listTables: ((database?) => Promise<string[]>);
    multiInsert: ((table, columns, values, useTransaction?) => Promise<QueryResult>);
    multiUpdate: ((table, primaryKey, columns, values, useTransaction?) => Promise<QueryResult>);
    prepareCreateIndex: ((table, fields, type) => Query[]);
    prepareCreateTable: ((name, fields, primaryKey, tableOptions?) => Query[]);
    prepareMultiInsert: ((table, columns, values) => Query[]);
    prepareMultiUpdate: ((table, primaryKey, columns, values) => Query[]);
    query: (<RecordType>(query, ...values) => Promise<QueryResult<RecordType>>);
    tableOptions: string;
    transaction: ((queries) => Promise<QueryResult[]>);
    truncate: ((tableOrTables, useTransaction) => Promise<boolean>);
    type: DatabaseType;
    typeName: string;
    use: ((database) => Promise<IDatabase>);
}
